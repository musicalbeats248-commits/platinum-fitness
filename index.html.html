const SHEET_NAME = 'Submissions';

function doGet(e) {
  const result = getSubmissions();
  return ContentService
    .createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  try {
    // Accept both JSON body and URL-encoded form data
    let submissions;
    try {
      const body = JSON.parse(e.postData.contents);
      submissions = body.submissions;
    } catch(ex) {
      submissions = JSON.parse(e.parameter.submissions);
    }
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    let sheet = ss.getSheetByName(SHEET_NAME) || ss.insertSheet(SHEET_NAME);
    sheet.getRange(1, 1).setValue(JSON.stringify(submissions || []));
    return ContentService
      .createTextOutput(JSON.stringify({ ok: true }))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({ ok: false, error: err.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function getSubmissions() {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    let sheet = ss.getSheetByName(SHEET_NAME) || ss.insertSheet(SHEET_NAME);
    const data = sheet.getRange(1, 1).getValue();
    const submissions = data ? JSON.parse(data) : [];
    return { ok: true, submissions };
  } catch(err) {
    return { ok: false, error: err.toString() };
  }
}
